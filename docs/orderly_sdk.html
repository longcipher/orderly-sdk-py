orderly_sdk

Orderly SDK for Python

Async REST and WebSocket client for Orderly.Network EVM API (v2).

This package provides:

  • AsyncClient: REST API client for trading and account management
  • OrderlyPublicWsManager: Public WebSocket client for market data
  • OrderlyPrivateWsManager: Private WebSocket client for account data
  • Custom exceptions for API errors
  • Logging utilities
Basic usage:
from orderly_sdk.rest import AsyncClient
from orderly_sdk.ws import OrderlyPublicWsManager

# REST API
client = AsyncClient(account_id=..., orderly_key=..., orderly_secret=..., endpoint=...)

# WebSocket API
ws_client = OrderlyPublicWsManager(account_id=..., endpoint=...)

For detailed documentation, see: https://longcipher.github.io/orderly-sdk-py/

 53    def __init__(
 54        self,
 55        _id="rest_client",

 89    async def close_connection(self):
 90        """
 91        Close the underlying aiohttp session.
 92        Call this when you are done with the client to free resources.
 93        """
 94        if self.session:
 95            assert self.session
 96            await self.session.close()

 98    async def _request(
 99        self,
100        method,
101        uri: str,
102        signed: bool,
103        params: Optional[Dict],
104        json: Optional[Dict],
105    ):
106        if signed:
107            ts = round(datetime.datetime.now().timestamp() * 1000)
108            json_str = ""
109            if json is not None:
110                logger.debug(f"request json body: {json}")
111                json_str = jsonlib.dumps(json)
112            path = urlparse(uri).path
113            if params:
114                query = urlencode(params)
115                path = f"{path}?{query}"
116            signature_str = f"{ts}{method.upper()}{path}{json_str}"
117            logger.debug(f"request signature: {signature_str}")
118            data_bytes = bytes(signature_str, "utf-8")
119
120            req_signature = base64.b64encode(
121                self.orderly_private_key.sign(data_bytes)
122            ).decode("utf-8")
123
124            self.headers["orderly-signature"] = req_signature
125            if self.account_id is not None:
126                self.headers["orderly-account-id"] = self.account_id
127            else:
128                pass
129            self.headers["orderly-key"] = f"ed25519:{self.orderly_key}"
130            self.headers["orderly-timestamp"] = str(ts)
131            self.headers["Content-Type"] = (
132                "application/json" if json else "application/x-www-form-urlencoded"
133            )
134            self.headers["Cache-Control"] = "no-cache"
135            self.session.headers.update(self.headers)
136
137        logger.debug("request uri: {}", uri)
138        async with getattr(self.session, method)(
139            uri, params=params, json=json
140        ) as response:
141            self.response = response
142            return await self._handle_response(response)
143
144    async def _handle_response(self, response: aiohttp.ClientResponse):
145        if not str(response.status).startswith("2"):
146            logger.error("response: {}", response)
147        try:
148            return await response.json()
149        except ValueError as exc:
150            txt = await response.text()
151            raise OrderlyRequestException(f"Invalid Response: {txt}") from exc
